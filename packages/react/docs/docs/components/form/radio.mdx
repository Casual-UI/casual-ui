### RadioGroup

```jsx live
function Demo() {
  const [val, setVal] = useState('Apple')

  const options = [
    { label: 'Apple', value: 'Apple' },
    { label: 'Banana', value: 'Banana' },
    { label: 'Peach', value: 'Peach' },
    { label: 'Grapes', value: 'Grapes' },
  ]

  return (
    <>
      <CRadioGroup
        value={val}
        onChange={setVal}
        options={options}
      />
      <div className="c-mt-md">Current selected: {val}</div>
    </>
  )
}
```
