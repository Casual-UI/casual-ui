### Basic

```jsx live
function Demo() {
  const options = [
    { label: '0 - 50', value: 0 },
    { label: '50 - 100', value: 1 },
    { label: '100 - 200', value: 2 },
    { label: '200 - 500', value: 3 },
    { label: '500 - 1000', value: 4 },
  ]
  const [selectedValue, setSelectedValue] = useState('')
  return (
    <CSelect
      value={selectedValue}
      onChange={setSelectedValue}
      placeholder="Please select"
      options={options}
    />
  )
}
```

function Demo() {
